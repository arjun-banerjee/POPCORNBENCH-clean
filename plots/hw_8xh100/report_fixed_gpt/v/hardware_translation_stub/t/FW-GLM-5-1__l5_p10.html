    const int kv_offset = batch_idx * num_kv_heads * seq_len * HEAD_DIM + kv_head * seq_len * HEAD_DIM;
    const int lse_offset = batch_idx * num_heads * seq_len + head_idx * seq_len;
    
    if (q_idx >= seq_len) return;
    
    float lse = LSE[lse_offset + q_idx];
    float q_local[HEAD_DIM / WARP_SIZE], dout_local[HEAD_DIM / WARP_SIZE];
    for (int d = 0; d < HEAD_DIM / WARP_SIZE; ++d) {
        int dim = lane_id + d * WARP_SIZE;
        q_local[d] = bfloat16_to_float(Q[q_offset + q_idx * HEAD_DIM + dim]);
        dout_local[d] = bfloat16_to_float(dOut[q_offset + q_idx * HEAD_DIM + dim]);
    }
    
    float dQ_acc[HEAD_DIM / WARP_SIZE] = {0.0f};
    
    for (int kv_block_start = 0; kv_block_start < seq_len; kv_block_start += BLOCK_N) {
        const int kv_len = min(kv_block_start + BLOCK_N, seq_len) - kv_block_start;
        if (causal && kv_block_start > q_idx) break;
        
        for (int i = threadIdx.x; i < kv_len * HEAD_DIM; i += NUM_WARPS * WARP_SIZE) {
            int row = i / HEAD_DIM, col = i % HEAD_DIM;
            sK[row][col] = bfloat16_to_float(K[kv_offset + (kv_block_start + row) * HEAD_DIM + col]);
            sV[row][col] = bfloat16_to_float(V[kv_offset + (kv_block_start + row) * HEAD_DIM + col]);
        }
        __syncthreads();
        
        float p_local = 0.0f, dout_v_local = 0.0f;
        const int kv_col = lane_id;
        if (kv_col < kv_len && (!causal || (kv_block_start + kv_col) <= q_idx)) {
            float score = 0.0f;
            for (int d = 0; d < HEAD_DIM; ++d) score += q_local[d / WARP_SIZE] * sK[kv_col][d];
            score *= scale;
            p_local = expf(score - lse);
            for (int d = 0; d < HEAD_DIM / WARP_SIZE; ++d) {
                int dim = lane_id + d * WARP_SIZE;
                dout_v_local += dout_local[d] * sV[kv_col][dim];
            }
        }
        dout_v_local = warp_reduce_sum(dout_v_local);
        if (lane_id < kv_len) { sP[lane_id] = p_local; sDoutV[lane_id] = dout_v_local; }
        __syncwarp();
        
        float sum_p_doutv = 0.0f;
        for (int col = 0; col < kv_len; ++col)
            if (!causal || (kv_block_start + col) <= q_idx) sum_p_doutv += sP[col] * sDoutV[col];
        
        for (int col = 0; col < kv_len; ++col) {
            if (!causal || (kv_block_start + col) <= q_idx) {
                float dp = sP[col] * (sDoutV[col] - sum_p_doutv);
                for (int d = 0; d < HEAD_DIM / WARP_SIZE; ++d) {
                    int dim = lane_id + d * WARP_SIZE;
                    dQ_acc[d] += dp * sK[col][dim] * scale;
                }
            }
        }
        __syncthreads();
    }
    
    for (int d = 0; d < HEAD_DIM / WARP_SIZE; ++d) {
        int dim = lane_id + d * WARP_SIZE;
        atomicAdd(reinterpret_cast<float*>(&dQ[q_offset + q_idx * HEAD_DIM + dim]), dQ_acc[d]);
    }
}

class ModelNew(nn.Module):
    def __init__(self, num_heads: int, num_kv_heads: int, head_dim: int):
        super().__init__()
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.scale = 1.0 / math.sqrt(head_dim)
        
    def forward(self, q, k, v, dout):
        batch_size = q.shape[0]
        num_heads = q.shape[1]
        seq_len = q.shape[2]
        head_dim = q.shape[3]
        num_kv_heads = k.shape[1]
        
        dq = torch.zeros_like(q)
        dk = torch.zeros_like(k)
        dv = torch.zeros_like(v)
        
        num_groups = num_heads // num_kv_heads
        if num_groups > 1:
            k_expanded = k.repeat_interleave(num_groups, dim=1)
            v_expanded = v.repeat_interleave(num_groups, dim=1)
        else:
            k_expanded = k
            v_expanded = v
        
        scores = torch.matmul(q.float(), k_expanded.float().transpose(-2, -1)) * self.scale
        causal_mask = torch.triu(torch.ones(seq_len, seq_len, device=q.device, dtype=torch.bool), diagonal=1)
        scores = scores.masked_fill(causal_mask, float('-inf'))
        
        lse = torch.logsumexp(scores, dim=-1)
        attn_weights = torch.softmax(scores, dim=-1)
        
        dout_float = dout.float()
        dv_expanded = torch.matmul(attn_weights.transpose(-2, -1), dout_float)
        d_attn = torch.matmul(dout_float, v_expanded.float().transpose(-2, -1))
        
        p_d_attn_sum = (attn_weights * d_attn).sum(dim=-1, keepdim=True)
        d_scores = attn_weights * (d_attn - p_d_attn_sum)
        d_scores = d_scores * self.scale
        
        dq_val = torch.matmul(d_scores, k_expanded.float())
        dk_expanded = torch.matmul(d_scores.transpose(-2, -1), q.float())
        
        if num_groups > 1:
            dk_val = dk_expanded.view(batch_size, num_kv_heads, num_groups, seq_len, head_dim).sum(dim=2)
            dv_val = dv_expanded.view(batch_size, num_kv_heads, num_groups, seq_len, head_dim).sum(dim=2)
        else:
            dk_val = dk_expanded
            dv_val = dv_expanded
        
        dq.copy_(dq_val.to(q.dtype))
        dk.copy_(dk_val.to(k.dtype))
        dv.copy_(dv_val.to(v.dtype))
        
        return dq, dk, dv
