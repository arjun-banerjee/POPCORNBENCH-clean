__device__ __forceinline__ int8_t float_to_int8_rn(float x) {
    uint32_t dst;
    asm volatile("cvt.rni.sat.s8.f32 %0, %1;" : "=r"(dst) : "f"(x));
    return reinterpret_cast<const int8_t&>(dst);
}

template <typename scalar_t>
__global__ void dynamic_scaled_int8_quant_kernel(
    const scalar_t* __restrict__ input, int8_t* __restrict__ output,
    float* scale_out, const int hidden_size) {
    const int tid = threadIdx.x;
    const int stride = blockDim.x;
    const int64_t token_idx = blockIdx.x;

    const scalar_t* row_in = input + token_idx * hidden_size;
    int8_t* row_out = output + token_idx * hidden_size;

    float thread_max = 0.f;
    for (int i = tid; i < hidden_size; i += stride) {
        float v = fabsf(static_cast<float>(row_in[i]));
        thread_max = fmaxf(thread_max, v);
    }
    __shared__ float sdata[256];
    sdata[tid] = thread_max;
    __syncthreads();
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) sdata[tid] = fmaxf(sdata[tid], sdata[tid + s]);
        __syncthreads();
    }
    float absmax = sdata[0];
    if (tid == 0) {
        scale_out[blockIdx.x] = (absmax == 0.f) ? 0.f : absmax / 127.f;
    }
    __syncthreads();
    float inv_s = (absmax == 0.f) ? 0.f : 127.f / absmax;
    for (int i = tid; i < hidden_size; i += stride) {
        row_out[i] = float_to_int8_rn(static_cast<float>(row_in[i]) * inv_s);
    }
}
