#include <torch/extension.h>
#include <cuda_fp16.h>
#include <cuda_runtime.h>

__global__ void attention_kernel(
    const half* __restrict__ q,
    const half* __restrict__ k,
    const half* __restrict__ v,
    half* __restrict__ out,
    int batch, int num_heads, int seq_len, int head_size, int num_kv_heads,
    float scale) {
    
    int b = blockIdx.x;
    int h = blockIdx.y;
    int kv_h = h / (num_heads / num_kv_heads);
    
    extern __shared__ half smem[];
    half* k_smem = smem;
    half* v_smem = smem + seq_len * head_size;
    
    int tid = threadIdx.x;
    
    // Load K and V into shared memory (simplified coalesced load)
    for (int i = tid; i < seq_len * head_size; i += blockDim.x) {
        int s = i / head_size;
        int d = i % head_size;
        k_smem[i] = k[((b * num_kv_heads + kv_h) * seq_len + s) * head_size + d];
        v_smem[i] = v[((b * num_kv_heads + kv_h) * seq_len + s) * head_size + d];
    }
    __syncthreads();
    
    // Compute attention for this head (naive but functional for small seq)
    for (int s_q = tid; s_q < seq_len; s_q += blockDim.x) {
        float max_val = -1e20f;
        float scores[1024];
        
        for (int s_k = 0; s_k < seq_len; s_k++) {
            float dot = 0.0f;
            for (int d = 0; d < head_size; d++) {
                float qv = __half2float(q[((b * num_heads + h) * seq_len + s_q) * head_size + d]);
                float kv = __half2float(k_smem[s_k * head_size + d]);
                dot += qv * kv;
            }
            scores[s_k] = dot * scale;
            max_val = fmaxf(max_val, scores[s_k]);
        }
        
        float exp_sum = 0.0f;
        for (int s_k = 0; s_k < seq_len; s_k++) {
            scores[s_k] = expf(scores[s_k] - max_val);
            exp_sum += scores[s_k];
        }
        
        for (int d = 0; d < head_size; d++) {
            float acc = 0.0f;
            for (int s_k = 0; s_k < seq_len; s_k++) {
                float w = scores[s_k] / exp_sum;
                acc += w * __half2float(v_smem[s_k * head_size + d]);
            }
            out[((b * num_heads + h) * seq_len + s_q) * head_size + d] = __float2half(acc);
        }
    }
}

torch::Tensor custom_attention(torch::Tensor q, torch::Tensor k, torch::Tensor v) {
    int batch = q.size(0);
    int num_heads = q.size(1);
    int seq_len = q.size(2);
    int head_size = q.size(3);
    int num_kv_heads = k.size(1);
    
    auto out = torch::empty_like(q);
    
    float scale = 1.0f / sqrtf((float)head_size);

}
