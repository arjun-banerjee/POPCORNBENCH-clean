#include <torch/extension.h>

#include <cuda_runtime.h>
#include <cutlass/cutlass.h>
#include <cutlass/gemm/device/gemm.h>
#include <cutlass/util/device_memory.h>

using ElementInputA = int8_t;
using ElementInputB = int8_t;
using ElementOutput = int32_t;
using ElementAccumulator = int32_t;
using ElementComputeEpilogue = int32_t;

using LayoutInputA = cutlass::layout::RowMajor;
using LayoutInputB = cutlass::layout::ColumnMajor;
using LayoutOutput = cutlass::layout::RowMajor;

using MMAOp = cutlass::arch::OpClassTensorOp;
using SmArch = cutlass::arch::Sm90;

using ShapeMMAThreadBlock = cutlass::gemm::GemmShape<128, 256, 64>;
using ShapeMMAWarp = cutlass::gemm::GemmShape<64, 64, 64>;
using ShapeMMAOp = cutlass::gemm::GemmShape<16, 8, 32>;

using SwizzleThreadBlock = cutlass::gemm::threadblock::GemmIdentityThreadblockSwizzle<>;

using EpilogueOp = cutlass::epilogue::thread::LinearCombination<
    ElementOutput, 4, ElementAccumulator, ElementComputeEpilogue>;

constexpr int NumStages = 3;

using Gemm = cutlass::gemm::device::Gemm<
    ElementInputA, LayoutInputA,
    ElementInputB, LayoutInputB,
    ElementOutput, LayoutOutput,
    ElementAccumulator,
    MMAOp, SmArch,
    ShapeMMAThreadBlock, ShapeMMAWarp, ShapeMMAOp,
    EpilogueOp, SwizzleThreadBlock, NumStages>;

torch::Tensor gemm_int8(torch::Tensor A, torch::Tensor B, int M, int N, int K, int32_t alpha, int32_t beta) {
    auto options = torch::TensorOptions().dtype(torch::kInt32).device(A.device());
    auto D = torch::zeros({M, N}, options);

    cutlass::gemm::GemmCoord problem_size(M, N, K);

    cutlass::TensorRef<ElementInputA, LayoutInputA> ref_A(
        reinterpret_cast<ElementInputA*>(A.data_ptr<int8_t>()), LayoutInputA(K));
    cutlass::TensorRef<ElementInputB, LayoutInputB> ref_B(
        reinterpret_cast<ElementInputB*>(B.data_ptr<int8_t>()), LayoutInputB(K));
    cutlass::TensorRef<ElementOutput, LayoutOutput> ref_C(
        reinterpret_cast<ElementOutput*>(D.data_ptr<int32_t>()), LayoutOutput(N));
    cutlass::TensorRef<ElementOutput, LayoutOutput> ref_D(
        reinterpret_cast<ElementOutput*>(D.data_ptr<int32_t>()), LayoutOutput(N));

    typename Gemm::Arguments arguments{
        problem_size,
        ref_A, ref_B, ref_C, ref_D,
        {alpha, beta},
        1
    };

    size_t workspace_size = Gemm::get_workspace_size(arguments);
    cutlass::device_memory::allocation<uint8_t> workspace(workspace_size);

    Gemm gemm_op;
    auto status = gemm_op.can_implement(arguments);
    if (status != cutlass::Status::kSuccess) {
        throw std::runtime_error("CUTLASS cannot implement");
    }
    status = gemm_op.initialize(arguments, workspace.get());

        throw std::runtime_error("CUTLASS initialize failed");
    }
    status = gemm_op();
    if (status != cutlass::Status::kSuccess) {
        throw std::runtime_error("CUTLASS launch failed");
    }
    return D;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("gemm_int8", &gemm_int8, "Hopper GEMM int8");
}
