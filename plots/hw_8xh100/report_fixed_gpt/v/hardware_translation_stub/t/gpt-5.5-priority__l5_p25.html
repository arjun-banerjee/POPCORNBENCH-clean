        Bs[ty][tx] = (br < K && col < N) ? __half2float(Bg[br * N + col]) : 0.0f;
        __syncthreads();
        #pragma unroll
        for (int t = 0; t < TILE; ++t) acc = fmaf(As[ty][t], Bs[t][tx], acc);
        __syncthreads();
    }
    if (row < M && col < N) C[((size_t)g * M + row) * N + col] = __float2half_rn(alpha * acc);
}

torch::Tensor grouped_gemm_forward(torch::Tensor A, torch::Tensor B, double alpha) {
    int G = A.size(0), M = A.size(1), K = A.size(2), N = B.size(2);
    auto C = torch::empty({G, M, N}, A.options().dtype(torch::kFloat16));
    dim3 block(TILE, TILE);
    dim3 grid((N + TILE - 1) / TILE, (M + TILE - 1) / TILE, G);
    grouped_gemm_kernel<<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(
        (const half*)A.data_ptr<at::Half>(), (const half*)B.data_ptr<at::Half>(),
        (half*)C.data_ptr<at::Half>(), G, M, K, N, (float)alpha);
    return C;
}
