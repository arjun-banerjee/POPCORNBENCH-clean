__device__ __forceinline__ float warp_reduce_max(float val) {
    #pragma unroll
    for (int mask = 16; mask > 0; mask >>= 1)
        val = fmaxf(val, __shfl_xor_sync(0xffffffff, val, mask));
    return val;
}

__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int mask = 16; mask > 0; mask >>= 1)
        val += __shfl_xor_sync(0xffffffff, val, mask);
    return val;
}

    const half* k_ptr = key + (batch_idx * num_kv_heads + kv_head) * seq_kv_len * HEAD_SIZE;
    const half* v_ptr = value + (batch_idx * num_kv_heads + kv_head) * seq_kv_len * HEAD_SIZE;
    half* o_ptr = out + (batch_idx * num_heads + head_idx) * seq_len * HEAD_SIZE;

            int row = i / HEAD_SIZE;
            int col = i % HEAD_SIZE;

            for (int h = 0; h < HEAD_SIZE; h++) {

            acc[2] += weight * h2f(V_row[lane_id + 64]);
            acc[3] += weight * h2f(V_row[lane_id + 96]);
        }
    }
    
    int out_row = q_start + q_row;
    if (sum_val > 0.0f) {
        float inv_sum = 1.0f / sum_val;
        o_ptr[out_row * HEAD_SIZE + lane_id] = f2h(acc[0] * inv_sum);
        o_ptr[out_row * HEAD_SIZE + lane_id + 32] = f2h(acc[1] * inv_sum);
        o_ptr[out_row * HEAD_SIZE + lane_id + 64] = f2h(acc[2] * inv_sum);

        for (int i = lane_id; i < HEAD_SIZE; i += WARP_SIZE) {
            o_ptr[out_row * HEAD_SIZE + i] = f2h(0.0f);
        }
    }
}
