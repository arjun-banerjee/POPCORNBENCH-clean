import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_fp16.h>
#include <cuda_runtime.h>

#define BLOCK_M 64
#define BLOCK_N 64
#define BLOCK_K 16
#define THREADS 256

__global__ void grouped_gemm_hopper_kernel(
    const half* __restrict__ A,
    const half* __restrict__ B,
    half* __restrict__ C,
    int groups, int M, int N, int K,
    float alpha, float beta
) {
    __shared__ half As[BLOCK_M][BLOCK_K];
    __shared__ half Bs[BLOCK_K][BLOCK_N];

    int group = blockIdx.z;
    int row_start = blockIdx.x * BLOCK_M;
    int col_start = blockIdx.y * BLOCK_N;

    const half* A_ptr = A + group * M * K;
    const half* B_ptr = B + group * K * N;
    half* C_ptr = C + group * M * N;

    int tx = threadIdx.x % 16;
    int ty = threadIdx.x / 16;

    for (int bk = 0; bk < K; bk += BLOCK_K) {
        // cooperative load A tile: each thread loads multiple
        for (int i = threadIdx.x; i < BLOCK_M * BLOCK_K; i += THREADS) {
            int r = i / BLOCK_K;
            int c = i % BLOCK_K;
            int gr = row_start + r;
            int gc = bk + c;
            As[r][c] = (gr < M && gc < K) ? A_ptr[gr * K + gc] : __float2half(0.0f);
        }
        for (int i = threadIdx.x; i < BLOCK_K * BLOCK_N; i += THREADS) {
            int r = i / BLOCK_N;
            int c = i % BLOCK_N;
            int gr = bk + r;
            int gc = col_start + c;
            Bs[r][c] = (gr < K && gc < N) ? B_ptr[gr * N + gc] : __float2half(0.0f);
        }
        __syncthreads();

        // each thread computes a 4x4 sub-tile or similar
        for (int m = ty; m < BLOCK_M; m += 16) {
            for (int n = tx; n < BLOCK_N; n += 16) {
                float acc = 0.0f;
                for (int k = 0; k < BLOCK_K; ++k) {
                    acc += __half2float(As[m][k]) * __half2float(Bs[k][n]);
                }
                int gm = row_start + m;
                int gn = col_start + n;
                if (gm < M && gn < N) {
                    float prev = (beta != 0.0f) ? __half2float(C_ptr[gm * N + gn]) : 0.0f;
                    C_ptr[gm * N + gn] = __float2half(alpha * acc + beta * prev);
                }
            }
        }
        __syncthreads();
    }
}

torch::Tensor grouped_gemm_cuda(torch::Tensor A, torch::Tensor B, int groups, float alpha, float beta) {
    auto C = torch::zeros({groups, A.size(1), B.size(2)}, torch::dtype(torch::kFloat16).device(A.device()));
    int M = A.size(1);
    int K = A.size(2);
    int N = B.size(2);
    dim3 grid((M + BLOCK_M - 1) / BLOCK_M, (N + BLOCK_N - 1) / BLOCK_N, groups);
    dim3 block(THREADS);
    grouped_gemm_hopper_kernel<<<grid, block>>>(
        reinterpret_cast<half*>(A.data_ptr<at::Half>()),
        reinterpret_cast<half*>(B.data_ptr<at::Half>()),
        reinterpret_cast<half*>(C.data_ptr<at::Half>()),
        groups, M, N, K, alpha, beta
    );
    return C;
}
