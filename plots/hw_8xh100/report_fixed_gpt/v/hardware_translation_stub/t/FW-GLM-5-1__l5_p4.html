#include <torch/all.h>
#include <cuda.h>
#include <cuda_fp16.h>
#include <ATen/cuda/CUDAContext.h>

// Hopper-optimized SWIGLU with half2 vectorization
// One block per token, threads process half2 pairs
__global__ void __launch_bounds__(512, 1) swiglu_kernel_hopper(
    c10::Half* __restrict__ output,
    const c10::Half* __restrict__ input,
    const int d) {
    
    const int token_idx = blockIdx.x;
    const int64_t base = (int64_t)token_idx * 2 * d;
    
    // Process as half2 pairs for vectorized memory access
    // Each half2 = 2 fp16 values
    const int num_pairs = d / 2;
    
    for (int i = threadIdx.x; i < num_pairs; i += blockDim.x) {
        // Load gate (first half) and up (second half) as half2
        const c10::Half* gate_base = input + base;
        const c10::Half* up_base = input + base + d;
        
        // Load 2 elements at a time
        const float gate0 = static_cast<float>(gate_base[2*i]);
        const float gate1 = static_cast<float>(gate_base[2*i + 1]);
        const float up0 = static_cast<float>(up_base[2*i]);
        const float up1 = static_cast<float>(up_base[2*i + 1]);
        
        // silu(gate) = gate / (1 + exp(-gate))
        const float silu0 = gate0 / (1.0f + expf(-gate0));
        const float silu1 = gate1 / (1.0f + expf(-gate1));
        
        // output = silu(gate) * up
        c10::Half* out_base = output + (int64_t)token_idx * d;
        out_base[2*i] = static_cast<c10::Half>(silu0 * up0);
        out_base[2*i + 1] = static_cast<c10::Half>(silu1 * up1);
    }
    
    // Handle odd d
    if ((d & 1) && threadIdx.x == 0) {
        const int last = d - 1;
        const float gate = static_cast<float>(input[base + last]);
        const float up = static_cast<float>(input[base + d + last]);
        const float silu = gate / (1.0f + expf(-gate));
        output[(int64_t)token_idx * d + last] = static_cast<c10::Half>(silu * up);
    }
}

void silu_and_mul_cuda(torch::Tensor output, torch::Tensor input) {
    const int d = input.size(-1) / 2;
    const int64_t num_tokens = input.numel() / input.size(-1);
    
    if (num_tokens == 0) return;
    
    const cudaStream_t stream = c10::cuda::getCurrentCUDAStream();
    
    // Choose block size based on d
    int threads = (d / 2 + 31) / 32 * 32;  // Round up to warp size
    threads = min(threads, 512);
    threads = max(threads, 32);
    
    dim3 grid(num_tokens);
    dim3 block(threads);
    
    swiglu_kernel_hopper<<<grid, block, 0, stream>>>(
        output.data_ptr<c10::Half>(),
        input.data_ptr<c10::Half>(),
        d
    );
}
