cuda_source = r"""
#include <cuda.h>
#include <cuda_fp16.h>
#include <cuda_runtime.h>

// Simple batched GEMM kernel with output permutation
// Tensor4DPermuteBMM0213: reshape [B, M, N] to [B/D1, D1, M, N] then permute to [B/D1, M, D1, N]
__global__ void __launch_bounds__(256, 8)
batched_gemm_permute_kernel(
    const __half* __restrict__ A,
    const __half* __restrict__ B,
    __half* __restrict__ D,
    int batch_size, int M, int N, int K,
    int D1,
    float alpha
) {
    int batch_idx = blockIdx.z;
    int m_idx = blockIdx.x * blockDim.y + threadIdx.y;
    int n_idx = blockIdx.y * blockDim.x + threadIdx.x;
    
    if (batch_idx >= batch_size || m_idx >= M || n_idx >= N) return;
    
    // Compute dot product
    const __half* A_row = A + batch_idx * M * K + m_idx * K;
    
    float sum = 0.0f;
    for (int k = 0; k < K; k++) {
        sum += __half2float(A_row[k]) * __half2float(B[k * N + n_idx]);
    }
    sum *= alpha;
    
    // Calculate output with permutation: [B/D1, M, D1, N]
    int batch_dim0 = batch_idx / D1;
    int batch_dim1 = batch_idx % D1;
    int out_idx = batch_dim0 * M * D1 * N + m_idx * D1 * N + batch_dim1 * N + n_idx;
    
    D[out_idx] = __float2half(sum);
}

extern "C" {
void launch_batched_gemm_permute(
    const void* A, const void* B, void* D,
    int batch_size, int M, int N, int K, int D1, float alpha,
    unsigned int bx, unsigned int by, unsigned int bz,
    unsigned int tx, unsigned int ty, unsigned int tz) {

}
}
