import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_fp16.h>
#include <cuda_runtime.h>

__global__ void gs_gemm_kernel(const half* __restrict__ A,
                               const half* __restrict__ B,
                               const int64_t* __restrict__ idx,
                               float* __restrict__ O,
                               int M, int K, int N, int IS) {
    int j = blockIdx.x;
    int m = blockIdx.y;
    int tid = threadIdx.x;
    // zero output cooperatively for first IS blocks per row covering all N
    for (int c = j * blockDim.x + tid; c < N; c += gridDim.x * blockDim.x) {
        O[m * N + c] = 0.0f;
    }
    __syncthreads();
    int col = (int)idx[j];
    float sum = 0.0f;
    // contiguous A over K, strided B by N.  This small fixed-K GEMM is latency tolerant
    // across many row/selected-column CTAs on H100.
    for (int k = tid; k < K; k += blockDim.x) {
        sum += __half2float(A[m * K + k]) * __half2float(B[k * N + col]);
    }
    __shared__ float smem[256];
    smem[tid] = sum;
    __syncthreads();
    if (tid < 128) smem[tid] += smem[tid + 128];
    __syncthreads();
    if (tid < 64) smem[tid] += smem[tid + 64];
    __syncthreads();
    if (tid < 32) {
        float v = smem[tid] + smem[tid + 32];
        #pragma unroll
        for (int off = 16; off > 0; off >>= 1) v += __shfl_down_sync(0xffffffff, v, off);
        if (tid == 0) O[m * N + col] = v;
    }
}

torch::Tensor gs_gemm(torch::Tensor A, torch::Tensor B, torch::Tensor idx) {
    const int M = (int)A.size(0);
    const int K = (int)A.size(1);
    const int N = (int)B.size(1);
    const int IS = (int)idx.size(0);
    auto O = torch::empty({M, N}, torch::TensorOptions().device(A.device()).dtype(torch::kFloat32));
    dim3 grid(IS, M);
    dim3 block(256);
    gs_gemm_kernel<<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(
        reinterpret_cast<const half*>(A.data_ptr<at::Half>()),
        reinterpret_cast<const half*>(B.data_ptr<at::Half>()),
        idx.data_ptr<int64_t>(),
        O.data_ptr<float>(), M, K, N, IS);
    return O;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("gs_gemm", &gs_gemm, "gather/scatter fused fp16 gemm");
