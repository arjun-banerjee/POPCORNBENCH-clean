template <int BLOCK_SIZE>
__global__ void attention_kernel(
    const half* __restrict__ q,
    const half* __restrict__ k,
    const half* __restrict__ v,
    half* __restrict__ out,
    const int batch,
    const int num_heads,
    const int seq_len,
    const int head_size,

    int b = blockIdx.x;
    int h = blockIdx.y;
    int tid = threadIdx.x;

    const half* q_ptr = q + (b * num_heads + h) * seq_len * head_size;
    const half* k_ptr = k + (b * num_heads + h) * seq_len * head_size;
    const half* v_ptr = v + (b * num_heads + h) * seq_len * head_size;
    half* out_ptr = out + (b * num_heads + h) * seq_len * head_size;

    for (int i = tid; i < seq_len * head_size; i += blockDim.x) {
        float sum = 0.0f;
