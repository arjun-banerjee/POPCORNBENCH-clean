#include <cuda_fp16.h>
#include <cuda_runtime.h>
#include <cub/cub.cuh>

template <int BLOCK_SIZE, int VEC_SIZE>
__global__ void rms_norm_kernel_h100(
    const half* __restrict__ input,
    const half* __restrict__ weight,
    half* __restrict__ output,
    const int hidden_size,
    const float epsilon,
    const int num_tokens) {

    __shared__ float s_variance;
    float variance = 0.0f;

    const int row = blockIdx.x;
    if (row >= num_tokens) return;

    const half* row_in = input + row * hidden_size;
    half* row_out = output + row * hidden_size;

    using float4_h = float4; // 128 bit
    const int vecs_per_row = hidden_size / 8; // VEC_SIZE=8 halfs

    for (int i = threadIdx.x; i < vecs_per_row; i += BLOCK_SIZE) {
        float4 v = reinterpret_cast<const float4*>(row_in)[i];
        half2* h2 = reinterpret_cast<half2*>(&v);
        #pragma unroll
        for (int j = 0; j < 4; j++) {
            float x = __half2float(h2[j].x);
            float y = __half2float(h2[j].y);
            variance += x * x + y * y;
        }
    }

    typedef cub::BlockReduce<float, BLOCK_SIZE> BlockReduce;
    __shared__ typename BlockReduce::TempStorage temp_storage;
    variance = BlockReduce(temp_storage).Reduce(variance, cub::Sum());

    if (threadIdx.x == 0) {
        s_variance = rsqrtf(variance / hidden_size + epsilon);
    }
    __syncthreads();

    const float inv_var = s_variance;
    const half* w_ptr = weight;

    for (int i = threadIdx.x; i < vecs_per_row; i += BLOCK_SIZE) {
        float4 v_in = reinterpret_cast<const float4*>(row_in)[i];
        float4 v_w = reinterpret_cast<const float4*>(w_ptr)[i];
        half2* h2_in = reinterpret_cast<half2*>(&v_in);
        half2* h2_w = reinterpret_cast<half2*>(&v_w);
        float4 v_out;
        half2* h2_out = reinterpret_cast<half2*>(&v_out);

        #pragma unroll
        for (int j = 0; j < 4; j++) {
            float x = __half2float(h2_in[j].x);

            float res_x = x * inv_var * w;

}
